// load sells
LOAD CSV WITH HEADERS FROM "https://storage.googleapis.com/all_tx/2/cc_sell.csv" AS value
MATCH (a:Account) where a.publicKey = value.TransactorPublicKeyBase58Check
MATCH (c:CCoin) where c.publicKey = value.CreatorPublicKey
CREATE (a)-[x:CCS_BURN {
  Height: value.Height,
  Index: value.Index,
  key: value.TransactionIDBase58Check + ":BURN",
  CreatorCoinAmountNanos: value.CreatorCoinAmountNanos,
  CreatorCoinAmountWhole: value.CreatorCoinAmountWhole
}]->(c)-[y:CCS_PAYOUT {
  Height: value.Height,
  Index: value.Index,
  key: value.TransactionIDBase58Check + ":PAYOUT",
  BitCloutNanosSoldFor: value.BitCloutNanosSoldFor,
  BitCloutWholeSoldFor: value.BitCloutWholeSoldFor
}]->(a);
